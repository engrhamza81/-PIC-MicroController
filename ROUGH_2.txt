// CONFIG2L
#pragma config PWRT = OFF       // Power-up Timer Enable bit (PWRT disabled)
#pragma config BOR = OFF        // Brown-out Reset Enable bit (Brown-out Reset disabled)
#pragma config BORV = 25        // Brown-out Reset Voltage bits (VBOR set to 2.5V)

// CONFIG2H
#pragma config WDT = OFF        // Watchdog Timer Enable bit (WDT disabled (control is placed on the SWDTEN bit))
#pragma config WDTPS = 128      // Watchdog Timer Postscale Select bits (1:128)

// CONFIG4L
#pragma config STVR = OFF       // Stack Full/Underflow Reset Enable bit (Stack Full/Underflow will not cause Reset)
#pragma config LVP = OFF        // Low-Voltage ICSP Enable bit (Low-Voltage ICSP disabled)

// CONFIG5L
#pragma config CP0 = OFF        // Code Protection bit (Block 0 (000200-001FFFh) not code protected)

// LCD pin definitions
#define ldata PORTB
#define rs LATE2
#define rw LATE1
#define en LATE0

// Function Prototypes
void lcdcmd(unsigned char);
void lcddata(unsigned char);
void lcd_init(void);
void delay(int);
unsigned int read_adc(void);
void display_temp(float);

void main(void) {
    // Setup
    TRISB = 0x00;       // LCD data lines as output
    TRISE = 0x00;       // LCD control lines as output
    TRISA0 = 1;         // RA0 (AN0) as input
    TRISD4 = 0;         // CCP1 (heater) output
    TRISC2 = 0;         // ECCP1 (fan) output
    AN0 = 1;

    // ADC configuration
    ADCON0 = 0b00000001; // Channel 0 (AN0), ADC ON
    ADCON1 = 0b00000000;

    // PWM configuration
    CCP1CON = 0b00001100;   // CCP1 PWM mode
    ECCP1CON = 0b00001100;  // ECCP1 PWM mode
    T2CON = 0b00000100;     // Timer2 ON, prescaler 1:1
    PR2 = 249;              // Set PWM frequency (~2kHz)

    TMR2ON = 1;             // Start Timer2

    lcd_init();             // Initialize LCD

    while (1) {
        unsigned int adc_val = read_adc();
        float voltage = adc_val * 5.0 / 1024.0;
        float temp = voltage * 100.0; // LM35 outputs 10mV per Â°C

        // Display on LCD
        display_temp(temp);

        // Fan/Heater control logic
        if (temp > 70.0) {
            ECCPR1H =ADRESH; // Fan ON
            CCPR1H = 0;
        } else if (temp < 70.0) {
            CCPR1H = ADRESH;  // Heater ON
            ECCPR1H = 0;
        } else {
            CCPR1H = 0;
            ECCPR1H = 0; // System OFF
        }

        delay(1000); // 1 second delay
    }
}

// Delay function
void delay(int ms) {
    for (int i = 0; i < ms; i++)
        for (int j = 0; j < 100; j++);
}

// ADC read function
unsigned int read_adc(void) {
    delay(2);
    ADCON0bits.GO = 1;
    while (ADCON0bits.GO);
    return (ADRESH);
}

// LCD Initialization
void lcd_init(void) {
    rw = 0;
    lcdcmd(0x38); // 8-bit mode, 2-line
    lcdcmd(0x0C); // Display on, cursor off
    lcdcmd(0x06); // Entry mode
    lcdcmd(0x01); // Clear display
    lcdcmd(0x80); // Set cursor at line 1
}

// Send command to LCD
void lcdcmd(unsigned char cmd) {
    ldata = cmd;
    rs = 0;
    en = 1;
    delay(2);
    en = 0;
}

// Send data to LCD
void lcddata(unsigned char data) {
    ldata = data;
    rs = 1;
    en = 1;
    delay(2);
    en = 0;
}

// Display temperature on LCD
void display_temp(float temp) {
    char str[16];

    lcdcmd(0x01);           // Clear LCD
    lcdcmd(0x80);           // Line 1
    sprintf(str, "Temp: %.1fC", temp);
    for (int i = 0; str[i] != '\0'; i++) {
        lcddata(str[i]);
    }

    lcdcmd(0xC0);           // Line 2
    if (temp > 30.0) {
        sprintf(str, "Fan ON       ");
    } else if (temp < 20.0) {
        sprintf(str, "Heater ON    ");
    } else {
        sprintf(str, "System OFF   ");
    }

    for (int i = 0; str[i] != '\0'; i++) {
        lcddata(str[i]);
    }
}
